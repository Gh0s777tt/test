name: Dependency Validation

on:
  schedule:
    # Run weekly on Monday at 6:00 AM UTC
    - cron: '0 6 * * 1'
  push:
    paths:
      - 'Cargo.toml'
      - 'Cargo.lock'
      - 'package.json'
      - 'package-lock.json'
      - 'requirements.txt'
      - 'pyproject.toml'
  pull_request:
    paths:
      - 'Cargo.toml'
      - 'Cargo.lock'
      - 'package.json'
      - 'package-lock.json'
      - 'requirements.txt'
      - 'pyproject.toml'
  workflow_dispatch:

permissions:
  contents: read

jobs:
  check-outdated:
    name: Check Outdated Dependencies
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Check Rust dependencies
        if: hashFiles('Cargo.toml') != ''
        run: |
          echo "### Rust Dependencies" >> $GITHUB_STEP_SUMMARY
          if command -v cargo &> /dev/null; then
            cargo install cargo-outdated --quiet 2>/dev/null || true
            cargo outdated --workspace || echo "Run \`cargo outdated\` locally for details"
          else
            echo "Rust not installed, skipping Cargo checks"
          fi

      - name: Check Node.js dependencies
        if: hashFiles('package.json') != ''
        run: |
          echo "### Node.js Dependencies" >> $GITHUB_STEP_SUMMARY
          if command -v npm &> /dev/null; then
            npm outdated || echo "Run \`npm outdated\` locally for details"
          else
            echo "Node.js not installed, skipping npm checks"
          fi

      - name: Check Python dependencies
        if: hashFiles('requirements.txt') != ''
        run: |
          echo "### Python Dependencies" >> $GITHUB_STEP_SUMMARY
          if command -v pip &> /dev/null; then
            pip list --outdated 2>/dev/null || echo "Run \`pip list --outdated\` locally for details"
          else
            echo "Python not installed, skipping pip checks"
          fi

  security-scan:
    name: Security Vulnerability Scan
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Run Trivy vulnerability scanner
        uses: aquasecurity/trivy-action@master
        with:
          scan-type: 'fs'
          scan-ref: '.'
          format: 'table'
          severity: 'CRITICAL,HIGH'
          exit-code: '0'
        continue-on-error: true

      - name: Check Rust vulnerabilities
        if: hashFiles('Cargo.lock') != ''
        run: |
          echo "### Rust Security Audit" >> $GITHUB_STEP_SUMMARY
          cargo install cargo-audit --quiet 2>/dev/null || true
          cargo audit 2>/dev/null || echo "Run \`cargo audit\` locally for security check"

      - name: Check Node.js vulnerabilities
        if: hashFiles('package-lock.json') != ''
        run: |
          echo "### Node.js Security Audit" >> $GITHUB_STEP_SUMMARY
          npm audit --audit-level=moderate || echo "Run \`npm audit\` locally for details"

  dependency-review:
    name: Dependency Review
    runs-on: ubuntu-latest
    if: github.event_name == 'pull_request'
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Dependency Review
        uses: actions/dependency-review-action@v4
        with:
          fail-on-severity: 'high'
          allow-licenses: 'MIT,Apache-2.0,BSD-2-Clause,BSD-3-Clause,ISC'

  license-check:
    name: License Compliance Check
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Check Rust licenses
        if: hashFiles('Cargo.lock') != ''
        run: |
          cargo install cargo-license --quiet 2>/dev/null || true
          cargo license --avoid-dev-deps 2>/dev/null > rust-licenses.txt || true
          echo "### Rust Dependencies Licenses" >> $GITHUB_STEP_SUMMARY
          cat rust-licenses.txt >> $GITHUB_STEP_SUMMARY 2>/dev/null || echo "No Rust dependencies found"

      - name: Check Node.js licenses
        if: hashFiles('package.json') != ''
        run: |
          npm install -g license-checker 2>/dev/null || true
          license-checker --csv licenses.csv 2>/dev/null || echo "No Node.js dependencies found"
          echo "### Node.js Dependencies Licenses" >> $GITHUB_STEP_SUMMARY
          head -20 licenses.csv >> $GITHUB_STEP_SUMMARY 2>/dev/null || true
